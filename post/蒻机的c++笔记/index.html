归并排序： 🔗

#include<iostream>
using namespace std;
const int N = 1e6+10;
int q[N],temp[N];
int n;
void merge_sort(int q[],int l, int r)
{
    if(l>=r) return ;
    int mid = (l+r)/2;// 以中间位置为界
    merge_sort(q,l,mid) , merge_sort(q,mid+1,r);//递归，把左半边和右半边分别排好序
    
    int k =0,i=l,j=mid+1;// i为左半边起点，j为右半边起点
    while(i<=mid && j<=r)
        if(q[i]<=q[j]) temp[k++] = q[i++];//把q[i]和q[j]中较小的放到temp[]中
        else temp[k++] = q[j++];
    while(i<=mid) temp[k++] = q[i++];//如果j走完了i还没走完就把i剩下的全部放到temp[]中
    while(j<=r) temp[k++] = q[j++];//如果i走完了j还没走完就把j剩下的全部放到temp[]中
    for(i = l,j=0;i<=r;i++,j++) q[i] = temp[j];//把q[]中的每一个用temp[]来替换
}
int main()
{
    cin >> n;
    for(int i =0;i<n;i++) cin >> q[i];
    merge_sort(q,0,n-1);
    for(int i=0;i<n;i++) cout << q[i] << " ";
}

#include<cstdio>
#include<iostream>
using namespace std;
char next(char x){
    return (x=='A')?'B':((x=='B')?'C':'A');
}
void hanoi(int n,char from,char to){
    if(n!=0){
        if(next(next(from))==to){
            char buffer=next(from);
            hanoi(n-1,from,to); // 把n-1个盘从A移到C
            printf("%d %c %c\n",n,from,buffer); // 表示把底下的盘从A移到B
            hanoi(n-1,to,from); // 把n-1个盘从C移到A
            printf("%d %c %c\n",n,buffer,to); // 表示把底下的盘从B移到C
            hanoi(n-1,from,to); // 把n-1个盘从A移到C
        }
	else{//B到C这种情况
            char buffer=next(next(from));
            hanoi(n-1,from,buffer); // 把n-1个盘从B移到A
            printf("%d %c %c\n" ,n,from,to); // 表示把底下的盘从B移到C
            hanoi(n-1,buffer,to); // 把n-1个盘从A移到C
        }
    }
}
int main(){
    int n;
    scanf("%d",&n);
    hanoi(n,'A','C');
}

{
    cin>>a1>>b1;
    int lena=a1.size(),lenb=b1.size(); 
    for(int i=1;i<=lena;i++) a[i]=a1[lena-i]-'0';  //倒着输入
    for(int j=1;j<=lenb;j++) b[j]=b1[lenb-j]-'0';  //倒着输入
    for(int i=1;i<=lenb;i++)
    {
        for(int j=1;j<=lena;j++)
        {
            c[i+j-1]+=a[j]*b[i];  //错位相加
        }
    }
    for(int i=1;i<=lena+lenb;i++)
    {
        if(c[i]>9)  //大于9就要进位
        {
            c[i+1]+=c[i]/10; 
            c[i]%=10;
        }
    }
    int lenc=lena+lenb;
    while(c[lenc]==0&&lenc>1) lenc--;  //去除前导零
    for(int i=lenc;i>=1;i--) cout<<c[i];
}

vector<int> add(vector<int>&A,vector<int>&B)
{
    vector<int> c;
    int t=0; //进位
    for(int i=0;i<A.size()||i<B.size();i++)
    {
        if(i<A.size()) t+=A[i];//A[i]这位有数 
        if(i<B.size()) t+=B[i];//A[i]+B[i]+t
        c.push_back(t%10);//取余数加到这一位
        t /=10;
    }
    if(t) c.push_back(1);
    return c;
}

#include<iostream>

// A/b 余数是r
vector<int> div(vector<int>&A,int b,int&r)
{
    vector<int>c;
    r=0;
    for(int i=A.size()-1;i>=0;i--) 
    {
        r = r*10 + A[i];
        c.push_back(r/b);
        r %=b;
    }
    reverse(c.begin(),c.end());
    while(c.size()>1 && c.back()==0) c.pop_back();
    return c;
}

高精度减法： 🔗

#include<iostream>
#include<vector>
using namespace std;

bool cmp(vector<int>&A,vector<int>&B)
{
    if(A.size()!=B.size()) return A.size()>B.size();//首先判断位数是否相同，谁的位数大谁就大
    for(int i=A.size()-1;i>=0;i--) // 从高位向低位一个一个比较大小
        if(A[i]!=B[i])
        return A[i]>B[i];
    return true;//两数相同
}
vector<int> sub(vector<int>&A,vector<int>&B)
{
    vector<int> c;
    for(int i=0,t=0;i<A.size();i++) //t表示每一位的值，表示借位
    {
        t = A[i]-t; // 原本 t = A[i]-B[i]-t; 需要判断B是否越界
        if(i<B.size()) t-=B[i];//B[i]是否存在
        c.push_back((t+10)%10); // 如果t>=0就是t本身，如果t<0就是t+10 
        if(t<0) t = 1; //小于0就要往前借一位
        else t = 0; // 不然的话就不用借位
    }
    // 逆着输入的如果是123-120 则vector存的结果是300
    // pop_back()删掉最后一位
    while(c.size()>1 && c.back()==0) c.pop_back(); //去掉前导0 300->30->3
    return c;
}

int main()
{
    string a,b;
    vector<int>A,B;
    cin >> a >> b;
    for(int i=a.size()-1;i>=0;i--) A.push_back(a[i]-'0');
    for(int i=b.size()-1;i>=0;i--) B.push_back(b[i]-'0');
    if(cmp(A,B))
    {
        auto c = sub(A,B); // auto = vector<int> auto 是让编译器自己判断类型
        for(int i=c.size()-1;i>=0;i--) cout << c[i];
    }
    else
    {
        cout << "-";
        auto c = sub(B,A);
        for(int i=c.size()-1;i>=0;i--) cout << c[i];
    }
   
}

#include<iostream>
using namespace std;
int main()
{
    int n,a[105],b[105],ans=0;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        b[i]=1;//自身的长度
        for(int j=1;j<i;j++)
        {
            if(a[j]<=a[i]&&b[j]>=b[i]) // 之前的数比现在的数小，之前的数对应的序列长度大于现在的数对应的序列长度
            b[i]=b[j]+1;
        }
        ans=max(ans,b[i]);
    }
    cout<<ans;
}

    int m,n;

    int r=m%n; //辗转相除法 // r = m-n 辗转相减法
    while(r>0)    
    {

#include<bits/stdc++.h>
using namespace std;
int n,a[15],book[15];
void dfs(int step) // step 表示第几个位置
{
    if(step==n+1)
    {
        for(int i=1;i<=n;i++) cout<<setw(5)<<a[i];
        cout<<endl;
        return ;  //返回上一级dfs
    }
    for(int i=1;i<=n;i++)
    {
        if(book[i]==0)  //如果这个数还没有被使用过
        {
            a[step] = i;
            book[i] = 1;
            dfs(step+1);
            book[i] = 0;  //把上一个返回的book[i]设为0
        }
    }
    return ; //退出dfs函数
}
